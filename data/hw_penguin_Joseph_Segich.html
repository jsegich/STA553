Make the graph

Bill_length=penguin$bill_length_mm
Flipper_length=penguin$flipper_length_mm
species=penguin$species
peng_size=(penguin$body_mass_g/3000)

## identifying the species
Adelie.id = which(species=="Adelie")  # value are case sensitive!
Gentoo.id = which(species=="Gentoo")
Chinstrap.id = which(species=="Chinstrap")

## making an empty plot: type = "n" ==> no point
plot(Bill_length,Flipper_length,xlab = ("Bill Length in Millimeters"), ylab = ("Flipper Length in Millimeters"), main = "Bill length vs Flipper length", type = "n")

points(Bill_length[Adelie.id], Flipper_length[Adelie.id], pch = 20, col = "red", cex=peng_size[Adelie.id])
points(Bill_length[Gentoo.id], Flipper_length[Gentoo.id], pch = 20, col = "blue", cex=peng_size[Gentoo.id])
points(Bill_length[Chinstrap.id], Flipper_length[Chinstrap.id], pch = 20, col = "cyan", cex=peng_size[Chinstrap.id])
legend("topleft", c("Adelie", "Gentoo", "Chinstrap"), 
                  col=c("red", "blue", "cyan"),
                  pch=c(20, 20, 20))
